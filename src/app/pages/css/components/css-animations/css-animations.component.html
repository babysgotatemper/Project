transition / transform

.box {
 width: 100px;
 height: 100px;
 background-color: #0000FF;
 transition: width 2s, height 2s, background-color 2s, transform 2s;
}
.box:hover {
 background-color: #FFCCCC;
 width: 200px;
 height: 200px;
 transform: rotate(180deg);
}
